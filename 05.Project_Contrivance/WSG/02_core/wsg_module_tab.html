									var ui = {
										init : function(){
											// Tab JS
											if ($('.js-tab .tab-nav a').length > 0){
												ui.tabs.init($('.js-tab .tab-nav a'));
											}
											// Tab Ajax
											if ($('.ajax-tab .tab-nav a').length > 0){
												ui.tabAjax.init($('.ajax-tab .tab-nav a'));
											}
										},
										// Tab JS
										tabs : {
											init : function($this){
												this.event($this);
											},
											event : function($this){
												// 현재페이지 탭 활성화
												$this.on('click', function(){
													ui.tabs.action($(this)); return false;
												});

												// 다른페이지 탭 활성화
												var tabPara = util.paramiter('tabId');
												if (tabPara != null){
													$this.each(function(){if ($(this).attr('href') == '#'+tabPara) ui.tabs.action($(this))})
												}
											},
											action : function($this){
												var target = $this.attr('href');
												$this.parent().addClass(setClass.active).siblings().removeClass(setClass.active);
												$(target).addClass(setClass.active).siblings().removeClass(setClass.active);
											},
										},
										// Tab Ajax
										tabAjax : {
											obj : '.ajax-tab',
											objActive : '.ajax-tab .tab-nav li'+ setClass.active +' a',
											objCont : '.tab-content',

											init : function($this){
												if ($(this.objActive).length > 0){this.action($(this.objActive))}
												this.event($this);
											},
											event : function($this){
												//현재페이지의 탭 활성화
												$this.on('click', function(){
													ui.tabAjax.action($(this)); return false;
												});

												//다른페이지의 탭 활성화
												var tabPara = util.paramiter('tabId');
												$this.each(function(){if ($(this).attr('href') == '#'+tabPara) ui.tabAjax.action($(this))})
											},
											action : function($this){
												var callTarget = $this.attr('href');
												var targetArea = $this.parents(this.obj).find(this.objCont);
												$this.parent().addClass(setClass.active).siblings().removeClass(setClass.active);
												targetArea.load(callTarget);
											},
										},
									}
